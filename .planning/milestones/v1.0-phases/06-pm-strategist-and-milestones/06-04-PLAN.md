```python
async def stream_to_discord(channel, stream_iter):
    msg = await channel.send("Thinking...")
    buffer = ""
    last_edit = time.monotonic()
    async for chunk in stream_iter:
        buffer += chunk
        if time.monotonic() - last_edit >= 1.0:
            await msg.edit(content=buffer[:2000])
            last_edit = time.monotonic()
    await msg.edit(content=buffer[:2000])
```
