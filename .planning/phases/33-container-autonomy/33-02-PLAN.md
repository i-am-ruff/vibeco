    **2. Rewrite NativeTransport** in `native.py`:

    The transport now spawns the worker with `--socket` flag, using `start_new_session=True` so the worker outlives the daemon. After spawning, it connects to the worker's socket.

    Socket path convention: `/tmp/vco-worker-{agent_id}.sock`

    ```python
    class NativeTransport:
        SOCKET_DIR = Path("/tmp")

        def __init__(self) -> None:
            self._processes: dict[str, asyncio.subprocess.Process] = {}

        def _socket_path(self, agent_id: str) -> Path:
            return self.SOCKET_DIR / f"vco-worker-{agent_id}.sock"

        async def spawn(
            self, agent_id: str, *, config: dict,
            env: dict[str, str] | None = None,
            working_dir: str | None = None,
        ) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            socket_path = self._socket_path(agent_id)
            cmd = [sys.executable, "-m", "vco_worker", "--socket", str(socket_path)]

            merged_env = dict(os.environ)
            if env:
                merged_env.update(env)
            merged_env["VCO_AGENT_ID"] = agent_id

            process = await asyncio.create_subprocess_exec(
                *cmd,
                stdin=asyncio.subprocess.DEVNULL,
                stdout=asyncio.subprocess.DEVNULL,
                stderr=asyncio.subprocess.PIPE,
                cwd=working_dir,
                env=merged_env,
                start_new_session=True,  # Survive daemon death
            )
            self._processes[agent_id] = process
            logger.info("Worker spawned for %s (pid=%d, socket=%s)", agent_id, process.pid or -1, socket_path)

            # Wait for socket to appear (worker needs time to bind)
            reader, writer = await self._wait_for_socket(agent_id, timeout=10.0)
            return reader, writer

        async def connect(self, agent_id: str) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            """Connect to an already-running worker's socket."""
            socket_path = self._socket_path(agent_id)
            if not socket_path.exists():
                raise ConnectionError(f"No socket for agent {agent_id} at {socket_path}")
            reader, writer = await asyncio.open_unix_connection(str(socket_path))
            logger.info("Connected to existing worker %s via %s", agent_id, socket_path)
            return reader, writer

        async def _wait_for_socket(self, agent_id: str, timeout: float = 10.0) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            """Wait for worker to create socket, then connect."""
            socket_path = self._socket_path(agent_id)
            deadline = asyncio.get_event_loop().time() + timeout
            while asyncio.get_event_loop().time() < deadline:
                if socket_path.exists():
                    try:
                        return await asyncio.open_unix_connection(str(socket_path))
                    except (ConnectionRefusedError, FileNotFoundError):
                        pass
                await asyncio.sleep(0.1)
            raise TimeoutError(f"Worker {agent_id} did not create socket within {timeout}s")

        async def terminate(self, agent_id: str) -> None:
            process = self._processes.pop(agent_id, None)
            socket_path = self._socket_path(agent_id)
            if process:
                process.terminate()
                try:
                    await asyncio.wait_for(process.wait(), timeout=5.0)
                except asyncio.TimeoutError:
                    process.kill()
            socket_path.unlink(missing_ok=True)

        @property
        def transport_type(self) -> str:
            return "native"
    ```

    **3. Rewrite DockerChannelTransport** in `docker_channel.py`:

    Use `docker run -d` (detached, no `-i`, no `--rm`). Mount a socket directory for communication. Worker inside container listens on socket, host connects.

    ```python
    class DockerChannelTransport:
        SOCKET_DIR = Path("/tmp/vco-sockets")

        def __init__(self, docker_image: str = "vco-agent:latest") -> None:
            self._image = docker_image
            self._containers: dict[str, str] = {}
            self.SOCKET_DIR.mkdir(parents=True, exist_ok=True)

        def _socket_path(self, agent_id: str) -> Path:
            return self.SOCKET_DIR / f"vco-worker-{agent_id}.sock"

        async def spawn(
            self, agent_id: str, *, config: dict,
            env: dict[str, str] | None = None,
            working_dir: str | None = None,
        ) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            container_name = f"vco-{agent_id}"
            socket_path = self._socket_path(agent_id)
            # Socket path inside container
            container_socket = f"/var/run/vco/vco-worker-{agent_id}.sock"

            cmd = [
                "docker", "run", "-d",
                "--name", container_name,
                "--network", "none",
            ]

            if env:
                for key, value in env.items():
                    cmd.extend(["-e", f"{key}={value}"])
            cmd.extend(["-e", f"VCO_AGENT_ID={agent_id}"])

            if working_dir:
                cmd.extend(["-v", f"{working_dir}:/workspace", "-w", "/workspace"])

            # Mount socket directory
            cmd.extend(["-v", f"{self.SOCKET_DIR}:/var/run/vco"])

            cmd.extend([
                self._image, "python", "-m", "vco_worker",
                "--socket", container_socket,
            ])

            logger.info("Spawning Docker container %s for agent %s", container_name, agent_id)

            proc = await asyncio.create_subprocess_exec(
                *cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
            stdout, stderr = await proc.communicate()
            if proc.returncode != 0:
                raise RuntimeError(f"docker run failed: {stderr.decode()}")
            container_id = stdout.decode().strip()

            self._containers[agent_id] = container_name
            logger.info("Docker container %s started for %s", container_name, agent_id)

            # Wait for socket to appear on host (via mount)
            reader, writer = await self._wait_for_socket(agent_id, timeout=15.0)
            return reader, writer

        async def connect(self, agent_id: str) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            """Connect to an already-running Docker worker's socket."""
            socket_path = self._socket_path(agent_id)
            if not socket_path.exists():
                raise ConnectionError(f"No socket for agent {agent_id} at {socket_path}")
            reader, writer = await asyncio.open_unix_connection(str(socket_path))
            logger.info("Connected to existing Docker worker %s", agent_id)
            return reader, writer

        async def _wait_for_socket(self, agent_id: str, timeout: float = 15.0) -> tuple[asyncio.StreamReader, asyncio.StreamWriter]:
            socket_path = self._socket_path(agent_id)
            deadline = asyncio.get_event_loop().time() + timeout
            while asyncio.get_event_loop().time() < deadline:
                if socket_path.exists():
                    try:
                        return await asyncio.open_unix_connection(str(socket_path))
                    except (ConnectionRefusedError, FileNotFoundError):
                        pass
                await asyncio.sleep(0.2)
            raise TimeoutError(f"Docker worker {agent_id} socket not ready within {timeout}s")

        async def terminate(self, agent_id: str) -> None:
            container_name = self._containers.pop(agent_id, None)
            if container_name:
                subprocess.run(["docker", "stop", container_name], capture_output=True, timeout=10)
                subprocess.run(["docker", "rm", "-f", container_name], capture_output=True, timeout=5)
            socket_path = self._socket_path(agent_id)
            socket_path.unlink(missing_ok=True)

        @property
        def transport_type(self) -> str:
            return "docker"
    ```

    async def stop_process(self, timeout: float = 10.0) -> None:
        """Gracefully stop the worker."""
        if not self.is_alive:
            return
        try:
            await self.send(StopMessage(reason="daemon shutdown"))
        except (RuntimeError, OSError):
            pass
        # Close socket if present
        if self._socket_writer is not None:
            if not self._socket_writer.is_closing():
                self._socket_writer.close()
            self._socket_writer = None
            self._socket_reader = None
        # Terminate process if present
        if self._process is not None and self._process.returncode is None:
            try:
                await asyncio.wait_for(self._process.wait(), timeout=timeout)
            except asyncio.TimeoutError:
                self._process.terminate()

                except asyncio.TimeoutError:
                    self._process.kill()
