```python
class MessagePriority(IntEnum):
    ESCALATION = 0      # Highest -- owner needs immediate attention
    SUPERVISOR = 1      # System-level alerts (degraded/recovered)
    STATUS = 2          # Routine state transitions
    HEALTH_DEBOUNCED = 3

@dataclass(order=True)
class QueuedMessage:
    priority: int
    timestamp: float
    channel_id: int = field(compare=False, default=0)
    content: str | None = field(compare=False, default=None)
    embed: discord.Embed | None = field(compare=False, default=None)

class MessageQueue:
    async def enqueue(self, msg: QueuedMessage) -> None: ...
    async def start(self) -> None: ...
    async def stop(self) -> None: ...

class RateLimited(Exception): ...
```
