       - async append(item) adds to end, persists
       - async insert_urgent(item) inserts at index 0, persists
       - async insert_after(after_id, item) inserts after matching item_id, raises ValueError if not found
       - async reorder(item_id, new_position) removes item by ID, re-inserts at min(new_position, len), persists
       - async cancel(item_id) sets status to CANCELLED, persists, raises ValueError if not found
       - async claim_next(agent_id) finds first PENDING item, sets status=ASSIGNED + assigned_to=agent_id, persists, returns item or None
       - async mark_completed(item_id) sets status to COMPLETED, persists, raises ValueError if not found
       - async mark_pending(item_id) sets status to PENDING, clears assigned_to, persists, raises ValueError if not found
       - property pending_items returns [i for i in self._items if i.status == PENDING]
