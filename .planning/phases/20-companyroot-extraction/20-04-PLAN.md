```python
"""Import boundary tests (EXTRACT-04).

Verifies that bot cog modules do not import container, supervisor, or agent
modules at runtime (only under TYPE_CHECKING guards).
"""
import ast
import pytest
from pathlib import Path

# Modules that the bot layer must NOT import at runtime
PROHIBITED_PREFIXES = [
    "vcompany.supervisor.company_root",
    "vcompany.supervisor.supervisor",
    "vcompany.supervisor.project_supervisor",
    "vcompany.container.container",
    "vcompany.container.child_spec",
    "vcompany.container.context",
    "vcompany.container.factory",
    "vcompany.resilience.message_queue",
    "vcompany.tmux.session",
]

# Files to check
BOT_FILES = [
    "src/vcompany/bot/client.py",
    "src/vcompany/bot/cogs/commands.py",
    "src/vcompany/bot/cogs/strategist.py",
]


def _get_runtime_imports(filepath: str) -> list[str]:
    """Extract module names from import statements NOT inside TYPE_CHECKING blocks."""
    with open(filepath) as f:
        tree = ast.parse(f.read())

    runtime_imports = []
    for node in ast.walk(tree):
        # Skip imports inside if TYPE_CHECKING blocks
        if isinstance(node, ast.If):
            test = node.test
            if isinstance(test, ast.Name) and test.id == "TYPE_CHECKING":
                continue
            if isinstance(test, ast.Attribute) and test.attr == "TYPE_CHECKING":
                continue

        if isinstance(node, ast.ImportFrom) and node.module:
            # Check if this import is inside a TYPE_CHECKING block
            # Simple heuristic: if it's at module level, it's runtime
            runtime_imports.append(node.module)

    return runtime_imports


def test_no_container_imports_in_bot():
    """Bot layer must not import container/supervisor modules at runtime."""
    violations = []
    for filepath in BOT_FILES:
        path = Path(filepath)
        if not path.exists():
            continue
        # Read file and check for prohibited imports outside TYPE_CHECKING
        with open(path) as f:
            content = f.read()
        for prefix in PROHIBITED_PREFIXES:
            # Check for "from {prefix} import" not inside TYPE_CHECKING
            if f"from {prefix} import" in content:
                # Verify it's not inside a TYPE_CHECKING block
                lines = content.split("\n")
                in_type_checking = False
                for line in lines:
                    stripped = line.strip()
                    if "TYPE_CHECKING" in stripped and stripped.startswith("if"):
                        in_type_checking = True
                    elif in_type_checking and not stripped.startswith((" ", "\t", "from", "import")):
                        in_type_checking = False
                    elif not in_type_checking and f"from {prefix} import" in stripped:
                        violations.append(f"{filepath}: {stripped}")

    assert not violations, f"Prohibited imports found:\n" + "\n".join(violations)


def test_no_discord_in_daemon():
    """Daemon package must not import discord."""
    daemon_dir = Path("src/vcompany/daemon")
    violations = []
    for py_file in daemon_dir.glob("*.py"):
        with open(py_file) as f:
            content = f.read()
        if "import discord" in content:
            violations.append(str(py_file))
    assert not violations, f"discord imports found in daemon: {violations}"
```
